```r
  last_digit = number %% 10

  number = (number - last_digit) / 10
  print(paste0("Number: ", number, " | Sum: ", sum, " | Last: ", last_digit))
}
```

```r
vector.info <- function(vector) {
  x <- list()
  x$Sum <- sum(vector)
  x$Mean <- mean(vector)
  x$Median <- median(vector)
  x$Min <- min(vector)
  x$Max <- max(vector)
  return(x)
}
```
